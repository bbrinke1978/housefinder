    ```typescript
    import { db } from "@/db/client";
    import { jvLeadMilestones } from "@/db/schema";
    import { logAudit } from "@/lib/audit-log";

    const QUALIFIED_CENTS = 1000;       // $10
    const ACTIVE_FOLLOW_UP_CENTS = 1500; // $15
    const DEAL_CLOSED_CENTS = 50000;    // $500

    /**
     * createQualifiedMilestone — fires when Brian accepts a JV lead.
     * Idempotent via UNIQUE(jv_lead_id, milestone_type).
     */
    export async function createQualifiedMilestone(
      jvLeadId: string,
      actorUserId: string | null
    ): Promise<{ created: boolean }> {
      const inserted = await db
        .insert(jvLeadMilestones)
        .values({
          jvLeadId,
          milestoneType: "qualified",
          amountCents: QUALIFIED_CENTS,
        })
        .onConflictDoNothing({ target: [jvLeadMilestones.jvLeadId, jvLeadMilestones.milestoneType] })
        .returning({ id: jvLeadMilestones.id });

      const created = inserted.length > 0;
      if (created) {
        await logAudit({
          actorUserId,
          action: "jv_lead.milestone_earned",
          entityType: "jv_lead",
          entityId: jvLeadId,
          newValue: { milestone: "qualified", amountCents: QUALIFIED_CENTS },
        });
      }
      return { created };
    }

    /** active_follow_up — fired by activity-actions.ts on first outbound contact_event. */
    export async function createActiveFollowUpMilestone(jvLeadId: string, actorUserId: string | null): Promise<{ created: boolean }>;

    /** deal_closed — fired by deal-actions.ts on status transition to 'closed'. */
    export async function createDealClosedMilestone(jvLeadId: string, actorUserId: string | null): Promise<{ created: boolean }>;
    ```
    Implement both remaining functions using the same template (different milestoneType + amountCents). Each returns `{ created: boolean }` so callers know whether to also fire the partner notification email (Plan 05 reads this).

    Why the `{ created }` return value: Plan 05's `notifyJvMilestoneEarned()` should ONLY fire when a milestone was actually inserted (not on duplicate calls). The boolean prevents re-notification on every contact_event re-log.

    DO NOT use a transaction wrapping insert + logAudit — `logAudit` is fire-and-forget and never throws; a transaction would defeat that property.
