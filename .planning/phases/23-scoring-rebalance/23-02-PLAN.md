```typescript
// Run directly: update hot_lead_threshold and ensure all signal types have correct weights
import { db, pool } from './src/db/client.js';
import { scraperConfig } from './src/db/schema.js';
import { eq } from 'drizzle-orm';

const NEW_THRESHOLD = {CHOSEN_THRESHOLD};  // substitute from checkpoint decision

const NEW_SIGNALS = [
  { signal_type: "nod",            weight: 3, freshness_days: 75  },
  { signal_type: "tax_lien",       weight: 2, freshness_days: 365 },
  { signal_type: "lis_pendens",    weight: 2, freshness_days: 180 },
  { signal_type: "probate",        weight: 1, freshness_days: 365 },
  { signal_type: "code_violation", weight: 1, freshness_days: 365 },
  { signal_type: "vacant",         weight: 1, freshness_days: 365 },
];

await db.update(scraperConfig).set({ value: String(NEW_THRESHOLD) }).where(eq(scraperConfig.key, 'hot_lead_threshold'));
await db.update(scraperConfig).set({ value: JSON.stringify(NEW_SIGNALS) }).where(eq(scraperConfig.key, 'scoring_signals'));
console.log('Config updated. Threshold:', NEW_THRESHOLD);
await pool.end();
```
