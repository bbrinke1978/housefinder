  <action>
Mirror the existing email-action pattern (look at `notifyLead` or `sendCampaignStep` or whatever's in this file).

```typescript
const BRIAN_EMAIL = "bbrinke1978@gmail.com"; // pull from env if BRIAN_EMAIL is already configured
const APP_BASE_URL = process.env.NEXT_PUBLIC_APP_URL ?? "https://finder.no-bshomes.com";

export async function notifyNewFeedbackItem(args: {
  itemId: string;
  type: "bug" | "feature" | "idea" | "question";
  title: string;
  description?: string | null;
  priority: "low" | "medium" | "high" | "critical";
  urlContext?: string | null;
  reporterName: string;
  reporterEmail: string;
}): Promise<void> {
  try {
    const deepLink = `${APP_BASE_URL}/feedback/${args.itemId}`;
    const html = await render(<FeedbackNewItemEmail {...args} deepLink={deepLink} />);
    await resend.emails.send({
      from: SENDER,
      to: BRIAN_EMAIL,
      subject: `[Feedback] ${args.type}: ${args.title}`,
      html,
    });
  } catch (err) {
    console.error("[notifyNewFeedbackItem] failed:", err);
    // intentionally swallow — don't fail the user's create action
  }
}

export async function notifyFeedbackShipped(args: {
  itemId: string;
  title: string;
  type: string;
  reporterEmail: string;
  reporterName: string;
  actorName: string;
  shipNote?: string;
}): Promise<void> {
  try {
    const deepLink = `${APP_BASE_URL}/feedback/${args.itemId}`;
    const html = await render(<FeedbackShippedEmail {...args} deepLink={deepLink} />);
    await resend.emails.send({
      from: SENDER,
      to: args.reporterEmail,
      subject: `[Feedback] Shipped: ${args.title}`,
      html,
    });
  } catch (err) {
    console.error("[notifyFeedbackShipped] failed:", err);
  }
}
```

Use whatever `render()` and `resend` and `SENDER` constants are already in this file.

// Fetch reporter's name + email for the email body (one query)
const [reporter] = await db.select({ name: users.name, email: users.email })
  .from(users).where(eq(users.id, session.user.id)).limit(1);

if (reporter) {
  // Fire-and-forget: don't await; let the action return to the user immediately
  notifyNewFeedbackItem({
    itemId: newId,
    type: input.type,
    title: input.title,
    description: input.description,
    priority: input.priority ?? "medium",
    urlContext: input.urlContext ?? null,
    reporterName: reporter.name,
    reporterEmail: reporter.email,
  }).catch((err) => console.error("[createFeedbackItem] email notify failed:", err));
}
```

The `.catch` on the unawaited promise is critical — Node would otherwise log an "unhandled promise rejection" warning.

if (newStatus === 'shipped') {
  // Skip self-notify — if Brian shipped his own report, no email needed
  if (item.reporterId === session.user.id) {
    console.log("[updateFeedbackStatus] self-notify skipped (reporter === actor)");
    return { success: true };
  }

  // Fetch reporter (the original) and the actor's name
  const [reporter] = await db.select({ name: users.name, email: users.email })
    .from(users).where(eq(users.id, item.reporterId)).limit(1);
  const [actor] = await db.select({ name: users.name })
    .from(users).where(eq(users.id, session.user.id)).limit(1);

  if (reporter && actor) {
    notifyFeedbackShipped({
      itemId: id,
      title: item.title,
      type: item.type,
      reporterEmail: reporter.email,
      reporterName: reporter.name,
      actorName: actor.name,
      shipNote,
    }).catch((err) => console.error("[updateFeedbackStatus] shipped email failed:", err));
  }
}
```
