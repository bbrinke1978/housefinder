export const alertHistory = pgTable("alert_history", {
  id: uuid("id").defaultRandom().primaryKey(),
  leadId: uuid("lead_id").notNull().references(() => leads.id),
  channel: text("channel").notNull(),
  runDate: text("run_date").notNull(),
  sentAt: timestamp("sent_at", { withTimezone: true }).notNull().defaultNow(),
});

export const leads = pgTable("leads", {
  id: uuid("id").defaultRandom().primaryKey(),
  propertyId: uuid("property_id").notNull().unique().references(() => properties.id),
  distressScore: integer("distress_score").notNull().default(0),
  isHot: boolean("is_hot").notNull().default(false),
  alertSent: boolean("alert_sent").notNull().default(false),
});

export const properties = pgTable("properties", {
  id: uuid("id").defaultRandom().primaryKey(),
  address: text("address").notNull(),
  city: text("city").notNull(),
  ownerName: text("owner_name"),
  ownerType: ownerTypeEnum("owner_type").default("unknown"),
});

export const scraperConfig = pgTable("scraper_config", {
  key: text("key").notNull().unique(),
  value: text("value").notNull(),
});

export const distressSignals = pgTable("distress_signals", {
  propertyId: uuid("property_id").notNull().references(() => properties.id),
  signalType: signalTypeEnum("signal_type").notNull(),
  status: signalStatusEnum("status").notNull().default("active"),
});

1. **scraper/src/alerts/index.ts** — Create the `sendAlerts` orchestrator:

   ```typescript
   export async function sendAlerts(context: InvocationContext): Promise<{ emailSent: number; smsSent: number }> {
   ```

   Logic:
   a. Read alert config from scraperConfig table:
      - `alerts.email.enabled` (default "true" if not found)
      - `alerts.sms.enabled` (default "true" if not found)
      - `alerts.email.threshold` (default "2" if not found)
      - `alerts.sms.threshold` (default "3" if not found)

   b. Get today's date: `const today = new Date().toISOString().split('T')[0]`

   c. Query hot leads NOT yet alerted today for email:
      - Join leads + properties
      - Where `leads.isHot = true` AND `leads.distressScore >= emailThreshold`
      - AND NOT EXISTS in alertHistory for (leadId, 'email', today)
      - Also join distressSignals to get signal types per lead (use a subquery or separate query)

   d. If emailEnabled and hot leads found:
      - Call `sendDigestEmail(hotLeads, appUrl)`
      - On success: insert alertHistory rows for each lead with channel='email', runDate=today
      - Log count via `context.log`

   e. Filter leads with score >= smsThreshold for SMS:
      - For each qualifying lead NOT yet SMS-alerted today (check alertHistory for 'sms' channel)
      - Call `sendSmsAlert(lead, appUrl)`
      - On success: insert alertHistory row with channel='sms'
      - Wrap each SMS send in try/catch — one failure should not block others

   f. Return `{ emailSent, smsSent }` counts

   Get `appUrl` from `process.env.APP_URL ?? 'https://housefinder.azurewebsites.net'`.

   Use `.js` extensions for all imports.
